use array

main {
  // Find files with shell — it's what shell is for
  let files = <<shell
    find . -name "*.csv" -type f
  >>

  // Parse CSV with Python — rich ecosystem
  let data = <<python[files]
    import csv, json
    rows = list(csv.DictReader(open(files.strip())))
    print(json.dumps(rows))
  >>

  // Crunch numbers with Julia — 100x faster math
  let stats = <<julia[data]
    using Statistics
    vals = parse.(Float64, split(data, ","))
    println("mean=$(mean(vals)) std=$(std(vals))")
  >>

  print("Results: " + stats)
}
